  % #2 is the vertical shift for inline placement
  \tikz[baseline=#2]{
    \filldraw (0,0) -- (0,#1) arc (90:(90-\insertframenumber/(\inserttotalframenumber)*360):#1);
    \draw (0,0) circle (#1);
  }
}
